import matplotlib.pyplot as plt
import seaborn as sns
# 设置中文字体
plt.rcParams['font.sans-serif'] = ['SimHei']

# 查看目标变量 'Outcome' 的分布比例
outcome_distribution_normalized = df['Outcome'].value_counts(normalize=True)
print("目标变量分布比例:\n", outcome_distribution_normalized)

# 使用柱状图可视化目标变量的分布
plt.figure(figsize=(8, 6))
sns.countplot(x='Outcome', data=df, hue='Outcome',palette='viridis',legend=False)
plt.title('目标变量分布-潘信宏')
plt.xticks([0, 1], ['未患病', '患病'])
# 在每个柱子上显示数值
heights = [rect.get_height() for rect in plt.gca().patches]
for i, v in enumerate(heights):
    plt.text(i, v + 0.5, f'{v}', horizontalalignment='center', verticalalignment='bottom')
plt.show()
