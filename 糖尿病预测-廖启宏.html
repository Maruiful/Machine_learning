from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score

#
from sklearn.metrics import f1_score
import matplotlib.pyplot as plt
import seaborn as sns

# 
LogisticRegression_model = LogisticRegression(max_iter=1000, random_state=42)

# 5F1
f1_scores = cross_val_score(LogisticRegression_model, X_train, y_train, cv=5, scoring='f1')

# F1
print(f"5F1: {f1_scores}")
print(f"F1: {f1_scores.mean()}")

# 
LogisticRegression_model.fit(X_train, y_train)

# 
y_train_pred = LogisticRegression_model.predict(X_train)
y_test_pred = LogisticRegression_model.predict(X_test)

# F1
f1_train = f1_score(y_train, y_train_pred)
f1_test = f1_score(y_test, y_test_pred)

print(f"F1: {f1_train}")
print(f"F1: {f1_test}")

# F1
plt.figure(figsize=(10, 6))

# 5F1
plt.subplot(1, 2, 1)
sns.barplot(x=['Fold 1', 'Fold 2', 'Fold 3', 'Fold 4', 'Fold 5'], 
            y=f1_scores, 
            hue=['Fold 1', 'Fold 2', 'Fold 3', 'Fold 4', 'Fold 5'], 
            palette='viridis',
            legend=False)
plt.title('5F1-')
plt.xlabel('')
plt.ylabel('F1')

# F1
plt.subplot(1, 2, 2)
sns.barplot(x=['', ''], 
            y=[f1_train, f1_test], 
            hue=['', ''], 
            palette='viridis',
            legend=False)
plt.title('F1-')
plt.xlabel('')
plt.ylabel('F1')

plt.tight_layout()
plt.show()
