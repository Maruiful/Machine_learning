### 对输入特征进行降维处理
from sklearn.decomposition import PCA
from sklearn import preprocessing                 # 调用标准化模块

x_cols = [col for col in df.columns if col!='Outcome']
y_col = 'Outcome'
X=df[x_cols].values
y=df[y_col].values

scaler = preprocessing.StandardScaler()           # 创建标准化对象
X_std = scaler.fit_transform(X)                   # 对数据进行标准化

pca = PCA(n_components=0.99)                      # 保留99%信息的主成分个主成分
X_pca = pca.fit(X_std).transform(X_std)           # 应用PCA进行降维

print("新的特征矩阵:\n", X_pca)

#引入逻辑回归和交叉验证的库
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score

#引入评价指标的库
from sklearn.metrics import f1_score
import matplotlib.pyplot as plt
import seaborn as sns

# 创建逻辑回归模型
LogisticRegression_model = LogisticRegression(max_iter=1000, random_state=42)

# 使用5折交叉验证评估模型的F1分数
f1_scores = cross_val_score(LogisticRegression_model, X_train, y_train, cv=5, scoring='f1')

# 打印交叉验证的F1分数
print(f"5折交叉验证F1分数: {f1_scores}")
print(f"交叉验证平均F1分数: {f1_scores.mean()}")

# 训练模型
LogisticRegression_model.fit(X_train, y_train)

# 进行预测
y_train_pred = LogisticRegression_model.predict(X_train)
y_test_pred = LogisticRegression_model.predict(X_test)

# 计算训练集和测试集的F1分数
f1_train = f1_score(y_train, y_train_pred)
f1_test = f1_score(y_test, y_test_pred)

print(f"训练集F1分数: {f1_train}")
print(f"测试集F1分数: {f1_test}")

# 可视化F1分数的结果
plt.figure(figsize=(10, 6))

# 第一张图：5折交叉验证F1分数
plt.subplot(1, 2, 1)
sns.barplot(x=['Fold 1', 'Fold 2', 'Fold 3', 'Fold 4', 'Fold 5'], 
            y=f1_scores, 
            hue=['Fold 1', 'Fold 2', 'Fold 3', 'Fold 4', 'Fold 5'], 
            palette='viridis',
            legend=False)
plt.title('5折交叉验证F1分数-廖启宏')
plt.xlabel('交叉验证折')
plt.ylabel('F1分数')

# 第二张图：训练集和测试集F1分数
plt.subplot(1, 2, 2)
sns.barplot(x=['训练集', '测试集'], 
            y=[f1_train, f1_test], 
            hue=['训练集', '测试集'], 
            palette='viridis',
            legend=False)
plt.title('训练集和测试集F1分数-廖启宏')
plt.xlabel('数据集')
plt.ylabel('F1分数')

plt.tight_layout()
plt.show()
