import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split

df = pd.read_csv(r'pima-indians-diabetes.data')

### 对输入特征进行降维处理
from sklearn.decomposition import PCA
from sklearn import preprocessing                 # 调用标准化模块

x_cols = [col for col in df.columns if col!='Outcome']
y_col = 'Outcome'
X=df[x_cols].values
y=df[y_col].values

scaler = preprocessing.StandardScaler()           # 创建标准化对象
X_std = scaler.fit_transform(X)                   # 对数据进行标准化

pca = PCA(n_components=0.99)                      # 保留99%信息的主成分个主成分
X_pca = pca.fit(X_std).transform(X_std)           # 应用PCA进行降维

X = X_pca
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.1, stratify=y, random_state=42)

实验改进 - 随机欠采样¶

从过采样的结果可以看出，模型的效果有了改善但仍然存在提升空间，尤其是XGBoost模型，过采样后模型效果提升不明显。因此我们决定继续探索其他方法，尝试提升模型效果。

我们选择尝试随机欠采样法，欠采样是从多数类中随机选择部分样本，使得两类样本数量更加均衡。这样可以使得模型在训练过程中不会过度偏向于多数类（未患病），从而提高对少数类（患病）的预测能力。

这里随机删除了测试集中的未患病样本，使两个样本的个数都为241。

from imblearn.under_sampling import RandomUnderSampler

# 对训练数据进行欠采样
X_train_undersampled, y_train_undersampled = undersampler.fit_resample(X_train, y_train)

# 查看欠采样后的类别分布
print('欠采样后的类别分布:', Counter(y_train_undersampled))

from sklearn.preprocessing import StandardScaler
from sklearn.feature_selection import SelectKBest, f_classif
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.svm import SVC
from sklearn.neighbors import KNeighborsClassifier
from xgboost import XGBClassifier

# 标准化特征
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train_undersampled)
X_test_scaled = scaler.transform(X_test)

# 特征选择
selector = SelectKBest(score_func=f_classif, k=8)
X_train_selected = selector.fit_transform(X_train_scaled, y_train_undersampled)
X_test_selected = selector.transform(X_test_scaled)

# 训练逻辑回归模型
lr_US_model = LogisticRegression(random_state=42, class_weight='balanced')
lr_US_model.fit(X_train_selected, y_train_undersampled)
y_US_pred_lr = lr_US_model.predict(X_test_selected)

# 训练随机森林模型
rf_US_model = RandomForestClassifier(random_state=42, class_weight='balanced')
rf_US_model.fit(X_train_selected, y_train_undersampled)
y_US_pred_rf = rf_US_model.predict(X_test_selected)

# 训练支持向量机模型
svc_US_model = SVC(random_state=42, class_weight='balanced', probability=True)
svc_US_model.fit(X_train_selected, y_train_undersampled)
y_pred_US_svc = svc_US_model.predict(X_test_selected)

# 训练K近邻模型
knn_US_model = KNeighborsClassifier(weights='distance')
knn_US_model.fit(X_train_selected, y_train_undersampled)
y_US_pred_knn = knn_US_model.predict(X_test_selected)

# 训练XGBoost模型
xgb_US_model = XGBClassifier(random_state=42, use_label_encoder=False, eval_metric='logloss', scale_pos_weight=500/268)
xgb_US_model.fit(X_train_selected, y_train_undersampled)
y_US_pred_xgb = xgb_US_model.predict(X_test_selected)

混淆矩阵¶

从混淆矩阵中可以看出，所有模型在未患病样本中的精确率都有提升，召回率有所降低；而在患病样本中的精确率都有降低但召回率有所提升。与未处理过的数据集相比，模型预测未患病人群的准确率有提升，但在患病人群中准确率有所降低。

from sklearn.metrics import confusion_matrix

undersalample_models = [
    lr_US_model,
    rf_US_model,
    svc_US_model,
    knn_US_model,
    xgb_US_model
]
models_names = ['逻辑回归', '随机森林', '支持向量机', 'K近邻', 'XGBoost']

plt.rcParams['font.sans-serif'] = ['SimHei']

# 创建3x2的子图布局
fig, axes = plt.subplots(nrows=3, ncols=2, figsize=(8, 12))

# 遍历所有模型生成混淆矩阵
for i, model in enumerate(undersalample_models):
    # 进行预测
    y_test_pred = model.predict(X_test)
    
    # 计算混淆矩阵
    conf_matrix = confusion_matrix(y_test, y_test_pred)
    
    # 获取子图的行和列位置
    row = i // 2
    col = i % 2
    
    # 绘制混淆矩阵
    sns.heatmap(conf_matrix, annot=True, fmt='d', cmap='Blues', xticklabels=['预测未患病', '预测患病'], yticklabels=['实际未患病', '实际患病'], ax=axes[row, col])
    axes[row, col].set_title(f'{models_names[i]}混淆矩阵-魏裕堃')
    axes[row, col].set_xlabel('预测')
    axes[row, col].set_ylabel('实际')

axes[2, 1].axis('off')
# 调整子图之间的间距
plt.tight_layout()
plt.show()

分类报告¶

通过分类报告给出的具体数据，我们可以得出以下结论：

  • 对于类别1（患病），模型召回率大幅提升，更擅长识别患病样本，减少漏诊风险。但精确率下降，有着更高的假阳性率。

  • 对于类别0（未患病），模型召回率降低，对未患病样本的识别能力下降，可能导致过度诊断。

  • 若任务更关注患病样本的检出率（如疾病筛查），欠采样后的模型更合适；若需平衡精确率和召回率，则需要探索其他方法。

展望¶

由上面的分析可以得出，欠采样通过减少多数类样本数量，显著提升了模型对少数类（患病）的识别能力，但代价是多数类（未患病）的性能下降。需根据实际需求（如医疗场景中更关注召回率）选择模型，并进一步通过阈值调整或集成方法优化性能。

对于进一步改进方向，可以考虑通过混合采样方法如‌SMOTE-Tomek、SMOTE-ENN或是更换数据集的方法来进一步提升模型的性能。
